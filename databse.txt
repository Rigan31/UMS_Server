CREATE TABLE Teacher_assigned(id uuid, teacher_id uuid, offered_course_id uuid, PRIMARY KEY(id));
CREATE TABLE Teacher_assigned_by_offered_course_id(id uuid, teacher_id uuid, offered_course_id uuid, PRIMARY KEY(offered_course_id, teacher_id));
CREATE TABLE Teacher_assigned_by_teacher_id(id uuid, teacher_id uuid, offered_course_id uuid, PRIMARY KEY(teacher_id, offered_course_id));


CREATE TABLE Evaluation_item(id uuid, name text, weight double, offered_course_id uuid, PRIMARY KEY(id));
CREATE TABLE Evaluation_item_by_offered_course_id(id uuid, name text, weight double, offered_course_id uuid, PRIMARY KEY(offered_course_id, id));


CREATE TABLE Evaluation_sub_item(id uuid, name text, mark double, evaluation_id uuid, PRIMARY KEY(id));
CREATE TABLE Evaluation_sub_item_by_evaluation_id(id uuid, name text, mark double, evaluation_id uuid, PRIMARY KEY(evaluation_id, id));
